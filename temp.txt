function startLoggingListener() {
    if (!firebaseActive) return;

    // limitToLast(5) ensures the client only downloads 5 items max, 
    // even if the database temporarily holds more records.
    db.ref('activity_log')
      .orderByChild('timestamp')
      .limitToLast(5)
      .on('value', (snapshot) => {
          const logsContainer = [];
          
          snapshot.forEach((childSnapshot) => {
              logsContainer.push({
                  id: childSnapshot.key,
                  ...childSnapshot.val()
              });
          });

          // Firebase returns limitToLast items in ascending order (oldest to newest).
          // We reverse it here so the absolute newest log appears at the top of your UI.
          const newestFirstLogs = logsContainer.reverse();
          
          // Execute your UI rendering logic here
          renderLogsToUI(newestFirstLogs); 
          
      }, (error) => {
          console.error("UI Logger Listener Failed:", error);
      });
}

        feed.prepend(entry); // Newest at the top
    });
}

// ==================== GLOBAL LOG (v2.4.0) ====================
function logGlobalEvent(message) {
    if (!firebaseActive || !auth.currentUser) return;
    
    console.log("📡 Attempting to log event:", message);
    
    const activityLogRef = db.ref('activity_log');
    const newLogRef = activityLogRef.push();
    
    // 1. Instantly save the new log entry
    newLogRef.set({
        text: message,
        timestamp: firebase.database.ServerValue.TIMESTAMP
    })
    .then(() => {
        // 2. Run memory management asynchronously in the background.
        // We trim down to 10 records instead of 5 to create a safe buffer zone,
        // preventing race conditions if multiple users are logging at once.
        return activityLogRef.orderByChild('timestamp').limitToLast(10).once('value');
    })
    .then((snapshot) => {
        if (!snapshot.exists()) return;

        // Collect all existing log IDs into an array
        const logKeys = [];
        snapshot.forEach((child) => {
            logKeys.push(child.key); // Automatically ordered oldest to newest
        });

        const maxBuffer = 5;
        if (logKeys.length > maxBuffer) {
            // Determine exactly how many overflow logs need to be deleted
            const keysToDelete = logKeys.slice(0, logKeys.length - maxBuffer);
            
            // Build a multi-path update object to delete everything in a single database request
            const updates = {};
            keysToDelete.forEach(key => {
                updates[key] = null; // Setting a node to null in Firebase deletes it
            });

            // Execute the batch deletion
            return activityLogRef.update(updates);
        }
    })
    .catch(err => console.error("Logger Database Error:", err));
}
